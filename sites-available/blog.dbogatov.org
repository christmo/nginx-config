server {
    server_name blog.dbogatov.org;

    access_log /var/log/nginx/blog.dbogatov.org.out.log;
    error_log /var/log/nginx/blog.dbogatov.org.err.log;

    ssl_certificate /etc/letsencrypt/live/blog.dbogatov.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/blog.dbogatov.org/privkey.pem;
    
    include /etc/nginx/snippets/ssl_config.conf;

    root /var/static/My-Blog/_site/;
    index index.html;
    
    try_files  $uri $uri.html $uri/index.html index.html;
    
    error_page 404 /404.html;
    error_page 500 502 503 504 /500.html; 

    include /etc/nginx/snippets/cache.conf;
    include /etc/nginx/snippets/deny_locations.conf;
    include /etc/nginx/snippets/letsencrypt.conf;
}
