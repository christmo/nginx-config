server {
    server_name music.dbogatov.org;

    access_log /var/log/nginx/music.dbogatov.org.out.log main_ext;
    error_log /var/log/nginx/music.dbogatov.org.err.log warn;
    
    ssl_certificate /etc/letsencrypt/live/dbogatov.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/dbogatov.org/privkey.pem;

    include /etc/nginx/snippets/ssl_config.conf;

    root /srv/www/php/sonerezh/app/webroot;
    index index.php;

    location / {
        alias /srv/www/php/sonerezh/app/webroot/;
        fastcgi_param PHP_VALUE open_basedir="/srv/www/php/sonerezh/:/mnt/hdd/music/";
        try_files $uri $uri/ /index.php?$args;
        expires 14d;
        add_header Cache-Control 'public';
    }

    location ~ \.php$ {
        alias /srv/www/php/sonerezh/app/webroot/$1;
        include fastcgi_params;
#       fastcgi_param CAKEPHP_DEBUG 1;
        fastcgi_split_path_info ^(.+\.php)(/.+)$;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param PATH_INFO $fastcgi_path_info;
        fastcgi_param modHeadersAvailable true; #Avoid sending the security headers twice
        fastcgi_pass unix:/run/php/php7.0-fpm.sock;
        try_files $uri =404;
    }

    include /etc/nginx/snippets/deny_locations.conf;
    include /etc/nginx/snippets/letsencrypt.conf;
}
